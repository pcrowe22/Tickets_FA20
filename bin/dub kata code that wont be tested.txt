function lookAndSay(data,len){
  // Populate result list with the look and say numbers
  // data:  starting number set
  // len:   sequence length
  var stringList = [];
  stringlist[0] = processLine(data);
  for (int a=1; a<=len; a++) {
    stringList[a] = processLine(stringList[a-1]);
  }
  return stringList;
  
  function processChar(line[i]) {
    var timesSeen = 0;
    var currentChar=line[i];
    while (currentChar==line[i]) {
      timesSeen++;
    }
    return String(timesSeen) + String(currentChar);
  }
  function processLine(line) {
    for (int i=0; i<line.length; i++) {
      return processChar(line[i]);
    }
  }
}
